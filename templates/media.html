#!/usr/bin/env python3 """ Module to provide and optionally write a simple HTML snippet containing an empty iframe and an embedded YouTube video. This module is designed to be production-ready with comprehensive error handling, logging, type hints, and clear documentation. Usage: python3 this_module.py # writes 'index.html' in the current directory python3 this_module.py -o out.html """ from __future__ import annotations import argparse import logging import sys from pathlib import Path from typing import Union # Constants should be UPPER_CASE DEFAULT_OUTPUT_FILENAME = "index.html" VIDEO_WIDTH = 560 VIDEO_HEIGHT = 315 VIDEO_ID = "ZGrdjGQZ9ug" VIDEO_QUERY = "si=8QpruIUflkEbyucK" HTML_TEMPLATE: str = f""" audio """ # Configure logging for the module LOG_FORMAT = "%(asctime)s - %(levelname)s - %(name)s - %(message)s" logging.basicConfig(level=logging.INFO, format=LOG_FORMAT) logger = logging.getLogger(__name__) def get_html() -> str: """ Return the HTML template string. Returns: str: The HTML content. """ logger.debug("Returning HTML template.") return HTML_TEMPLATE def resolve_output_path(file_path: Union[str, Path]) -> Path: """ Resolve and validate the provided file_path argument. Args: file_path (Union[str, Path]): Path to validate and resolve. Returns: pathlib.Path: Resolved Path object. Raises: TypeError: If file_path is not a string or pathlib.Path. ValueError: If file_path is None or resolves to an empty path. """ if file_path is None: logger.error("Provided file_path is None.") raise ValueError("file_path must not be None") if not isinstance(file_path, (str, Path)): logger.error("file_path has invalid type: %s", type(file_path)) raise TypeError("file_path must be a str or pathlib.Path") out_path = Path(file_path).expanduser().resolve() if str(out_path).strip() == "": logger.error("Resolved file path is empty after stripping.") raise ValueError("file_path must not be empty") return out_path def ensure_parent_dir(path: Path) -> None: """ Ensure that the parent directory of the provided path exists, creating it if necessary. Args: path (pathlib.Path): Path whose parent directory should be present. Returns: None """ parent_dir = path.parent if not parent_dir.exists(): logger.debug(f"Creating parent directories for path: {parent_dir}") parent_dir.mkdir(parents=True, exist_ok=True) def write_html_file(file_path: Union[str, Path]) -> Path: """ Write the HTML_TEMPLATE to the given file path using safe file operations. Args: file_path (Union[str, Path]): Path to the output file. Returns: Path: Path object pointing to the written file. Raises: TypeError: If file_path is not a string or pathlib.Path. ValueError: If file_path resolves to an empty path. OSError: For filesystem related errors (propagated after logging). """ out_path = resolve_output_path(file_path) try: # Ensure parent directory exists ensure_parent_dir(out_path) # Write file using 'with' context manager and explicit encoding logger.info(f"Writing HTML content to {out_path}") with out_path.open("w", encoding="utf-8") as f: f.write(get_html()) logger.info(f"Successfully wrote HTML to {out_path}") return out_path except OSError as exc: logger.exception(f"Failed to write HTML to {out_path}: {exc}") raise def parse_args(argv: list[str] | None = None) -> argparse.Namespace: """ Parse command-line arguments. Args: argv (list[str] | None): Argument list to parse (defaults to sys.argv[1:]). Returns: argparse.Namespace: Parsed arguments namespace with attribute 'output'. """ if argv is None: argv = sys.argv[1:] parser = argparse.ArgumentParser( description="Write a simple HTML file containing an empty iframe and an embedded YouTube video." ) parser.add_argument( "-o", "--output", default=DEFAULT_OUTPUT_FILENAME, help=f"Output HTML file (default: {DEFAULT_OUTPUT_FILENAME})", ) return parser.parse_args(argv) def main(argv: list[str] | None = None) -> int: """ Main entry point for command-line execution. This function parses arguments and writes the HTML file to the specified location. Args: argv (list[str] | None): Optional list of command-line arguments. Returns: int: Exit code (0 for success, non-zero for failure). """ try: args = parse_args(argv) output_path = args.output written_path = write_html_file(output_path) logger.info(f"HTML written to: {written_path}") return 0 except Exception as exc: # pragma: no cover - top-level exception handling logger.exception("An error occurred: %s", exc) return 1 if __name__ == "__main__": raise SystemExit(main())
