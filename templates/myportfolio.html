This module provides a single function `generate_portfolio_html` that writes a corrected and accessible portfolio HTML page to disk using UTF-8 encoding. Comprehensive error handling and logging are included. Usage: from portfolio_writer import generate_portfolio_html generate_portfolio_html("portfolio.html") Or run as a script: python portfolio_writer.py --output portfolio.html """ from __future__ import annotations import argparse import logging from pathlib import Path from typing import Optional, Union # Configure module-level logger logger = logging.getLogger(__name__) logger.setLevel(logging.INFO) _handler = logging.StreamHandler() _handler.setFormatter( logging.Formatter("%(asctime)s - %(name)s - %(levelname)s - %(message)s") ) logger.addHandler(_handler) DEFAULT_OUTPUT_FILE = "portfolio.html" DEFAULT_ENCODING = "utf-8" IMAGE_DIMENSION = 200 TABLE_BORDER = 2 # HTML content to be written. """ def _validate_output_path(output: Union[str, Path]) -> Path: """ Validate and normalize the output path. Accepts either a string or a pathlib.Path and returns a resolved Path object. """ if isinstance(output, Path): path = output elif isinstance(output, str): path = Path(output) else: raise TypeError("output must be a str or pathlib.Path") # Expand user and resolve, but do not require that parent exists. try: resolved = path.expanduser().resolve() except Exception: # If resolution fails (e.g., path includes non-existing mount), fallback to expanduser. resolved = path.expanduser() return resolved def generate_portfolio_html( output_file: Union[str, Path] = DEFAULT_OUTPUT_FILE, encoding: str = DEFAULT_ENCODING, ) -> Path: """ Generate the portfolio HTML file at the given output_file path using the specified encoding. Returns the Path to the written file. Raises exceptions for invalid inputs. """ if output_file is None: raise ValueError("output_file must not be None") if encoding is None: raise ValueError("encoding must not be None") if not isinstance(encoding, str): raise TypeError("encoding must be a str") out_path = _validate_output_path(output_file) html_content = PORTFOLIO_HTML.format(image_dim=IMAGE_DIMENSION, table_border=TABLE_BORDER) # Ensure parent directory exists parent = out_path.parent if parent and not parent.exists(): try: parent.mkdir(parents=True, exist_ok=True) logger.debug("Created parent directory: %s", parent) except Exception as exc: # pragma: no cover - defensive logger.error("Could not create parent directory %s: %s", parent, exc) raise try: with out_path.open("w", encoding=encoding) as fh: fh.write(html_content) logger.info("Portfolio HTML written to %s", out_path) except Exception as exc: # pragma: no cover - defensive logger.error("Failed to write portfolio HTML to %s: %s", out_path, exc) raise return out_path def _parse_args(argv: Optional[list[str]] = None) -> argparse.Namespace: """ Parse command-line arguments. Returns the populated Namespace object. """ parser = argparse.ArgumentParser(description="Write a portfolio HTML file.") parser.add_argument( "--output", "-o", default=DEFAULT_OUTPUT_FILE, help=f"Output file path (default: {DEFAULT_OUTPUT_FILE})", ) parser.add_argument( "--encoding", "-e", default=DEFAULT_ENCODING, help=f"File encoding (default: {DEFAULT_ENCODING})", ) return parser.parse_args(argv) def main(argv: Optional[list[str]] = None) -> int: """ Entry point for CLI usage. Parses arguments and calls generate_portfolio_html. Returns exit code 0 on success. """ args = _parse_args(argv) try: output_path = generate_portfolio_html(args.output, args.encoding) logger.info("Successfully created portfolio at %s", output_path) return 0 except Exception as exc: # pragma: no cover - defensive logger.exception("An error occurred: %s", exc) return 1 if __name__ == "__main__": # pragma: no cover - invoked as script raise SystemExit(main())
