""" def _build_portfolio_html(image_dim: int = IMAGE_DIMENSION, table_border: int = TABLE_BORDER) -> str: """Build the final HTML by injecting constants into the template. """ return PORTFOLIO_HTML.format(image_dim=image_dim, table_border=table_border) def _write_text_to_file(path: Path, text: str, encoding: str = DEFAULT_ENCODING) -> Path: """Write text to a file using the provided encoding and return the Path. This helper uses a context manager to ensure the file is correctly closed and logs any exceptions that occur during writing. Parameters: path: The filesystem path where the content will be written. text: The textual content to write. encoding: The encoding to use when writing the file. Returns: The Path object pointing to the written file. Raises: OSError: If an I/O error occurs when writing the file. """ try: logger.info("Writing portfolio HTML to '%s' with encoding '%s'.", str(path), encoding) # Use binary-safe Path.open with text mode and explicit encoding with path.open("w", encoding=encoding) as f: f.write(text) except OSError: logger.exception("Failed to write portfolio HTML to '%s'.", str(path)) raise return path def generate_portfolio_html(output: Union[str, Path] = DEFAULT_OUTPUT_FILE, encoding: Optional[str] = None) -> Path: """Generate the portfolio HTML file and write it to disk.
