""" def _coerce_path(output: Union[str, Path, None]) -> Path: """Coerce the given output to a pathlib.Path instance. Accepts None, str, or Path and returns a Path pointing to the destination file. """ if output is None: return Path(DEFAULT_OUTPUT_FILE) if isinstance(output, Path): return output if isinstance(output, str): return Path(output) raise TypeError("output must be a str, pathlib.Path, or None") def generate_portfolio_html(output: Union[str, Path] = DEFAULT_OUTPUT_FILE) -> Path: """Write the portfolio HTML to the given output path. The function will create or overwrite the file at `output` using UTF-8 encoding. It returns the Path to the written file. Args: output: Destination file path as a string or pathlib.Path. """ out_path = _coerce_path(output) try: html = PORTFOLIO_HTML.format(image_dim=IMAGE_DIMENSION, table_border=TABLE_BORDER) # Ensure parent directory exists if out_path.parent != Path("."): out_path.parent.mkdir(parents=True, exist_ok=True) with out_path.open("w", encoding=DEFAULT_ENCODING) as fh: fh.write(html) logger.info(f"Wrote portfolio HTML to {out_path}") return out_path except Exception as exc: # pragma: no cover - top-level safe guard logger.exception("Failed to write portfolio HTML: %s", exc) raise def _parse_args(argv: list[str] | None = None) -> argparse.Namespace: """Parse command-line arguments and return the namespace. """ args = _parse_args(argv) try: out = generate_portfolio_html(args.output) logger.info(f"Portfolio generated at: {out}") return 0 except Exception: logger.error("An error occurred while generating the portfolio HTML.") return 1 if __name__ == "__main__": raise SystemExit(main())
