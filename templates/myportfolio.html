""" def generate_portfolio_html(output_path: Union[str, Path]) -> Path: """ Write the portfolio HTML content to a file using UTF-8 encoding. OSError: If an OS-related error occurs while writing the file. """ if output_path is None: logger.error("No output_path provided to generate_portfolio_html.") raise ValueError("output_path must not be None") if not isinstance(output_path, (str, Path)): logger.error( "Invalid type for output_path: %s (expected str or Path).", type(output_path), ) raise ValueError("output_path must be a string or pathlib.Path") path = Path(output_path) try: # Ensure parent directory exists if path.parent and not path.parent.exists(): logger.debug("Creating parent directories for: %s", path.parent) path.parent.mkdir(parents=True, exist_ok=True) # Write the file using a context manager and UTF-8 encoding with path.open("w", encoding="utf-8") as fh: fh.write(PORTFOLIO_HTML) logger.info("Portfolio HTML written successfully to: %s", path.resolve()) return path.resolve() except OSError as exc: logger.exception("Failed to write portfolio HTML to %s: %s", path, exc) raise def _parse_args() -> argparse.Namespace: """Parse command-line arguments for the script.""" parser = argparse.ArgumentParser(description="Write a portfolio HTML file.") parser.add_argument( "--output", "-o", default=DEFAULT_OUTPUT_FILE, help=f"Path to output HTML file (default: {DEFAULT_OUTPUT_FILE})", ) return parser.parse_args() def main() -> None: """Main entrypoint for CLI usage.""" args = _parse_args() try: output_path = generate_portfolio_html(args.output) logger.info("Done.
